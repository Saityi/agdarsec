NList : Set ‚Üí ‚Ñï ‚Üí Set
NList A zero    = A
NList A (suc n) = List (NList A n)

module _ {Chars : ‚Ñï ‚Üí Set} {{ùïä : Sized Char Chars}} where

 NList‚Ä≤ : {A : Set} ‚Üí [ Parser Char Chars Maybe A ] ‚Üí
          (n : ‚Ñï)   ‚Üí [ Parser Char Chars Maybe (NList A n) ]
 NList‚Ä≤ A zero    = A
 NList‚Ä≤ A (suc n) = parens $ box $ DList.toList <$>
                    chainl1 (DList.[_] <$> NList‚Ä≤ A n) (box $ DList._++_ <$ char ',')
